 function DistOf(px,py, p1x,p1y :Integer):Integer;

    p1x,p1y - координаты второй точки
    result - расстояние между первой и второй точкой.

 function DateAdd(Part: TDatePart; Value: Integer; const Date: TDate): TDate;
